sync_thehive_closures(app_state):
```python
async def sync_thehive_closures(app_state) -> None:
    try:
        client = app_state.thehive_client
        if client is None:
            return
        sqlite_conn = app_state.stores.sqlite._conn
        resolved = await client.find_resolved_cases()
        for case in resolved:
            thehive_id = case.get("_id", "")
            status = case.get("resolutionStatus", "Indeterminate")
            end_date_ms = case.get("endDate")
            closed_at = None
            if end_date_ms:
                from datetime import datetime, timezone
                closed_at = datetime.fromtimestamp(end_date_ms / 1000, tz=timezone.utc).isoformat()
            analyst = case.get("assignee", "")
            await asyncio.to_thread(
                _update_detection_thehive_status,
                sqlite_conn, thehive_id, status, closed_at, analyst
            )
    except Exception as exc:
        log.warning("TheHive closure sync failed: %s", exc)
```

drain_pending_cases(sqlite_conn, thehive_client):
```python
async def drain_pending_cases(sqlite_conn, thehive_client) -> None:
    rows = await asyncio.to_thread(_get_pending_cases, sqlite_conn)
    for row in rows:
        row_id = row["id"]
        detection_id = row["detection_id"]
        attempts = row["attempts"]
        if attempts >= 5:
            continue  # Abandon after 5 failures
        try:
            payload = json.loads(row["payload_json"])
            created = await thehive_client.create_case(payload["case"])
            case_id = created["_id"]
            case_num = created.get("number") or created.get("caseId")
            for obs in payload.get("observables", []):
                try:
                    await thehive_client.create_observable(case_id, obs)
                except Exception:
                    pass
            await asyncio.to_thread(_save_thehive_case_id, sqlite_conn, detection_id, case_id, case_num, "New")
            await asyncio.to_thread(_delete_pending_case, sqlite_conn, row_id)
        except Exception as exc:
            await asyncio.to_thread(_increment_pending_attempts, sqlite_conn, row_id, str(exc))
```
