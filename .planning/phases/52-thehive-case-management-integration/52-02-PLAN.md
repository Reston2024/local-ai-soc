For _maybe_create_thehive_case():
```python
async def _maybe_create_thehive_case(detection: dict, sqlite_conn, thehive_client) -> None:
    severity = detection.get("severity", "").lower()
    if severity not in ("high", "critical"):
        return
    rule_id = detection.get("rule_id", "")
    if rule_id in settings.THEHIVE_SUPPRESS_RULES:
        return
    try:
        case_payload = build_case_payload(detection)
        created = await thehive_client.create_case(case_payload)
        case_id = created["_id"]
        case_num = created.get("number") or created.get("caseId")
        for obs in build_observables(detection):
            try:
                await thehive_client.create_observable(case_id, obs)
            except Exception as exc:
                log.warning("Failed to add observable: %s", exc)
        await asyncio.to_thread(_save_thehive_case_id, sqlite_conn, detection["id"], case_id, case_num, "New")
    except Exception as exc:
        log.warning("TheHive case creation failed, queuing for retry: %s", exc)
        payload_json = json.dumps({
            "case": build_case_payload(detection),
            "observables": build_observables(detection),
        })
        await asyncio.to_thread(_enqueue_pending_case, sqlite_conn, detection["id"], payload_json)
```
