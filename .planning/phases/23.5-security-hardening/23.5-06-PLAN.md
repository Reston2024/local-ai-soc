```python
def verify_totp(secret: str, code: str, operator_id: str) -> bool:
    """
    Verify a 6-digit TOTP code against the operator's secret.

    Replay prevention uses two layers:
    1. L1 (in-memory): _seen_totp dict — fast, cleared on restart
    2. L2 (SQLite system_kv): persists across restarts — authoritative

    A code rejected by either layer is a replay.
    """
    import time
    kv_key = f"totp_seen_{operator_id}_{code}"

    # L1: in-memory check
    if _seen_totp.get(operator_id) == code:
        return False

    # L2: SQLite check (if store is available)
    if _sqlite_store is not None:
        try:
            stored = _sqlite_store.get_kv(kv_key)
            if stored is not None:
                # Found in SQLite — replay
                _seen_totp[operator_id] = code  # populate L1 too
                return False
        except Exception:
            pass  # SQLite unavailable — fall through to pyotp check

    # Verify the code
    totp_obj = pyotp.TOTP(secret)
    if totp_obj.verify(code, valid_window=1):
        # Mark as seen in both layers
        _seen_totp[operator_id] = code
        if _sqlite_store is not None:
            try:
                expire_ts = str(int(time.time()) + _TOTP_TTL_SECONDS)
                _sqlite_store.set_kv(kv_key, expire_ts)
            except Exception:
                pass  # Non-fatal: in-memory L1 is still active
        return True

    return False
```

async def _write_telemetry(
    self,
    *,
    model: str,
    endpoint: str,
    prompt_chars: int,
    completion_chars: int,
    latency_ms: int,
    success: bool,
    error_type: Optional[str] = None,
    prompt_text: Optional[str] = None,
    prompt_hash: Optional[str] = None,
) -> None:
    if self._duckdb_store is None:
        return
    import uuid
    from datetime import datetime, timezone

    # Truncate prompt_text to 64KB max to prevent unbounded storage
    _MAX_PROMPT_TEXT = 65536
    stored_prompt = None
    if prompt_text is not None:
        stored_prompt = prompt_text[:_MAX_PROMPT_TEXT]

    # Compute SHA-256 hash of full prompt for integrity (use full text, not truncated)
    stored_hash = prompt_hash
    if stored_hash is None and prompt_text is not None:
        stored_hash = hashlib.sha256(prompt_text.encode("utf-8", errors="replace")).hexdigest()

    await self._duckdb_store.execute_write(
        """INSERT OR IGNORE INTO llm_calls
           (call_id, called_at, model, endpoint, prompt_chars,
            completion_chars, latency_ms, success, error_type,
            prompt_text, prompt_hash)
           VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)""",
        [
            str(uuid.uuid4()),
            datetime.now(timezone.utc).isoformat(),
            model,
            endpoint,
            prompt_chars,
            completion_chars,
            latency_ms,
            success,
            error_type,
            stored_prompt,
            stored_hash,
        ],
    )
