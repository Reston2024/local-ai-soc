```python
def test_default_token_rejected():
    """Settings must raise ValueError if AUTH_TOKEN is 'changeme' or shorter than 32 chars."""
    import pytest
    from pydantic import ValidationError
    from backend.core.config import Settings

    # Default value must be rejected
    with pytest.raises((ValueError, ValidationError)):
        Settings(AUTH_TOKEN="changeme")

    # Explicit dev bypass value is accepted (allows intentional weak tokens for local dev)
```

```python
@model_validator(mode="after")
def reject_default_auth_token(self) -> "Settings":
    """Reject startup if AUTH_TOKEN is the default 'changeme' or is weak.

    Raises ValueError so pydantic wraps it in ValidationError at Settings()
    construction time — never at request time.

    The value 'dev-only-bypass' is explicitly allowed for local development
    without a .env file. All other values shorter than 32 characters are rejected.
    """
    token = self.AUTH_TOKEN
    _ALLOWED_WEAK = {"dev-only-bypass"}
    if token == "changeme":
        raise ValueError(
            "AUTH_TOKEN is set to the default 'changeme'. "
            "Generate a strong token: python -c \"import secrets; print(secrets.token_hex(32))\" "
            "and set AUTH_TOKEN=<token> in your .env file."
        )
    if token not in _ALLOWED_WEAK and len(token) < 32:
        raise ValueError(
            f"AUTH_TOKEN is too short ({len(token)} chars). "
            "Minimum 32 characters required. "
            "Generate: python -c \"import secrets; print(secrets.token_hex(32))\""
        )
    return self
```

IMPORTANT: The module-level `settings = Settings()` call at the bottom of config.py will now
raise ValidationError if AUTH_TOKEN is "changeme" in the environment. This is intentional —
it prevents the application from starting with a default token. Tests that import config.py
without providing a valid token must patch Settings before import or use environment override.

For the test, pydantic-settings reads from environment. Call `Settings(AUTH_TOKEN=...)` with
keyword argument to bypass .env during test. The ValidationError is raised by pydantic wrapping

```python
# --- Legacy AUTH_TOKEN fallback ---
if hmac.compare_digest(raw, configured):
    # Require TOTP even on legacy path — no backdoor admin without MFA
    legacy_secret = getattr(settings, "LEGACY_TOTP_SECRET", "").strip()
    if not legacy_secret:
        # Legacy path disabled: no LEGACY_TOTP_SECRET configured
        raise HTTPException(
            status_code=401,
            detail="Legacy admin path disabled: configure LEGACY_TOTP_SECRET to use this path",
        )

    if not totp_code:
        raise HTTPException(
            status_code=401,
            detail="TOTP code required for legacy admin path (X-TOTP-Code header)",
        )
    from backend.core.totp_utils import verify_totp
    if not verify_totp(legacy_secret, totp_code, "legacy-admin"):
        raise HTTPException(status_code=401, detail="Invalid or replayed TOTP code")

        operator_id="legacy-admin",
        username="admin",

    )
```
