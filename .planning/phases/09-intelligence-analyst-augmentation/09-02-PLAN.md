```python
def save_investigation(
    self,
    detection_id: str,
    graph_snapshot: dict,
    metadata: dict,
) -> str:
    """Persist an investigation snapshot. Returns the new investigation ID."""
    import json, uuid
    from datetime import datetime, timezone
    inv_id = uuid.uuid4().hex
    created_at = datetime.now(timezone.utc).isoformat()
    self._conn.execute(
        "INSERT INTO saved_investigations (id, detection_id, graph_snapshot, metadata, created_at) "
        "VALUES (?, ?, ?, ?, ?)",
        (inv_id, detection_id, json.dumps(graph_snapshot), json.dumps(metadata), created_at),
    )
    self._conn.commit()
    return inv_id

def list_saved_investigations(self) -> list[dict]:
    """Return all saved investigation records (newest first)."""
    import json
    cursor = self._conn.execute(
        "SELECT id, detection_id, metadata, created_at FROM saved_investigations "
        "ORDER BY created_at DESC"
    )
    rows = cursor.fetchall()
    return [
        {
            "id": row[0],
            "detection_id": row[1],
            "metadata": json.loads(row[2] or "{}"),
            "created_at": row[3],
        }
        for row in rows
    ]

def get_saved_investigation(self, investigation_id: str) -> dict | None:
    """Return a single saved investigation by ID, or None if not found."""
    import json
    cursor = self._conn.execute(
        "SELECT id, detection_id, graph_snapshot, metadata, created_at "
        "FROM saved_investigations WHERE id = ?",
        (investigation_id,),
    )
    row = cursor.fetchone()
    if row is None:
        return None
    return {
        "id": row[0],
        "detection_id": row[1],
        "graph_snapshot": json.loads(row[2] or "{}"),
        "metadata": json.loads(row[3] or "{}"),
        "created_at": row[4],
    }
```
