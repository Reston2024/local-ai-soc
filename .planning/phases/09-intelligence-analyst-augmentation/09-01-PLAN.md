MITRE_WEIGHTS: dict[str, int] mapping technique IDs to point values:
- Critical techniques (T1003.001 LSASS dump, T1071.001 C2 web protocol) -> 40 pts each
- High techniques (T1547.001 registry persistence, T1059.001 PowerShell, T1055 process injection) -> 30 pts each
- Medium techniques (T1033 system owner discovery, T1087.002 AD account enum, T1083 file discovery) -> 20 pts each
- Low techniques (T1057 process discovery, T1012 registry query) -> 10 pts each

SEVERITY_BASE: dict mapping severity string to base points:
- "critical" -> 40, "high" -> 30, "medium" -> 20, "low" -> 10, "info" -> 5

def score_detection(severity: str, technique_id: str | None, anomaly_count: int) -> int:
  Additive components: SEVERITY_BASE[severity] + MITRE_WEIGHTS.get(technique_id, 0) + min(anomaly_count * 10, 20)
  Return min(total, 100)

def score_entity(entity_id: str, events: list[dict], detections: list[dict], anomaly_flags: list[str]) -> int:
  Component 1 (0-40): max MITRE weight across all events' attack_technique fields
  Component 2 (0-30): max severity base across all events
  Component 3 (0-20): min(len(anomaly_flags) * 10, 20)
  Component 4 (0-10): min(len(detections) * 5, 10)
  Return min(sum of components, 100)

def enrich_nodes_with_risk_score(nodes: list[dict], scored_entities: dict[str, int]) -> list[dict]:
  For each node in nodes, set node["data"]["risk_score"] = scored_entities.get(node["data"]["id"], 0).
  Return nodes (mutate in place is fine, same list reference).

```python
from dataclasses import dataclass
from typing import Callable

_OFFICE_APPS = {"winword.exe", "excel.exe", "powerpnt.exe", "outlook.exe", "mspub.exe"}
_SHELL_PROCS = {"cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe", "mshta.exe", "rundll32.exe"}
_SYSTEM_PROCS = {"svchost.exe", "lsass.exe", "csrss.exe", "smss.exe", "winlogon.exe"}
_MASQUERADE_PATHS = {"appdata", "temp", "tmp", "programdata\\microsoft\\windows\\start menu"}
_STANDARD_PORTS = {80, 443, 8080, 8443, 53, 22, 21, 25, 587, 110, 143, 3389, 445, 139}
_RFC1918 = [("10.", 8), ("172.16.", 12), ("192.168.", 16), ("127.", 8)]

@dataclass
class AnomalyRule:
    rule_id: str
    name: str
    description: str
    check: Callable[[dict], bool]

def _is_external_ip(ip: str) -> bool:
    if not ip:
        return False
    for prefix, _ in _RFC1918:
        if ip.startswith(prefix):
            return False
    return True

ANOMALY_RULES: list[AnomalyRule] = [
    AnomalyRule(
        rule_id="ANO-001",
        name="office_spawns_shell",
        description="Office application spawning a shell or scripting process",
        check=lambda evt: (
            evt.get("parent_process_name", "").lower() in _OFFICE_APPS
            and evt.get("process_name", "").lower() in _SHELL_PROCS
        ),
    ),
    AnomalyRule(
        rule_id="ANO-002",
        name="system_process_unusual_parent",
        description="System process spawned by a non-system, non-services parent",
        check=lambda evt: (
            evt.get("process_name", "").lower() in _SYSTEM_PROCS
            and evt.get("parent_process_name", "").lower() not in
            (_SYSTEM_PROCS | {"services.exe", "wininit.exe", ""})
        ),
    ),
    AnomalyRule(
        rule_id="ANO-003",
        name="process_masquerading",
        description="Process with system-like name running from unexpected path (AppData/Temp)",
        check=lambda evt: (
            evt.get("process_name", "").lower() in _SYSTEM_PROCS
            and any(
                p in (evt.get("process_path", "") or "").lower()
                for p in _MASQUERADE_PATHS
            )
        ),
    ),
    AnomalyRule(
        rule_id="ANO-004",
        name="unusual_external_port",
        description="Connection to external IP on non-standard port (potential C2)",
        check=lambda evt: (
            _is_external_ip(evt.get("dest_ip", ""))
            and evt.get("dest_port") is not None
            and int(evt.get("dest_port", 0)) not in _STANDARD_PORTS
        ),
    ),
]

def check_event_anomalies(event: dict) -> list[str]:
    """Return list of rule_ids that fire for this event."""
    return [r.rule_id for r in ANOMALY_RULES if r.check(event)]
```
