    ```typescript
    hunts: {
      query: (query: string, analyst_id = 'analyst') =>
        request<HuntResult>('/api/hunts/query', {
          method: 'POST',

          body: JSON.stringify({ query, analyst_id }),

    },

      get: (ip: string) =>
        request<OsintResult>(`/api/osint/${ip}`, { headers: authHeaders() }),
    },
    ```
