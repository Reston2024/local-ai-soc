```python
from fastapi import Depends, HTTPException, Request
from fastapi.security import HTTPBearer
from typing import Callable

def require_role(*allowed_roles: str) -> Callable:
    """
    FastAPI dependency factory enforcing role-based access.

    Usage:
        @router.post("/api/operators", dependencies=[Depends(require_role("admin"))])
        async def create_operator(...): ...

    Returns the OperatorContext so routes can use it if needed:
        @router.get("/api/operators")
        async def list_operators(ctx: OperatorContext = Depends(require_role("admin", "analyst"))): ...

    Raises:
        HTTPException(401): propagated from verify_token (no valid token)
        HTTPException(403): valid token but role not in allowed_roles
    """
    from backend.core.auth import verify_token  # deferred to avoid circular import

    async def _dep(
        request: Request,
        ctx: OperatorContext = Depends(verify_token),
    ) -> OperatorContext:
        if ctx.role not in allowed_roles:
            raise HTTPException(
                status_code=403,
                detail=f"Forbidden: requires role in {list(allowed_roles)}, got '{ctx.role}'",
            )
        return ctx

    return _dep
```

Note: verify_token is imported inside the closure (deferred) to avoid circular
import between auth.py and rbac.py (auth.py imports OperatorContext from rbac.py).
