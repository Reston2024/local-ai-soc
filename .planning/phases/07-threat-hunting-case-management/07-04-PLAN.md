    class CaseUpdateRequest(BaseModel):

        tags: Optional[list[str]] = None

    **POST /api/cases** — create case:
    - sqlite = request.app.state.stores.sqlite
    - case_id = await asyncio.to_thread(_case_manager.create_investigation_case, sqlite, body.title, body.description)
    - case = await asyncio.to_thread(sqlite.get_investigation_case, case_id)
    - return {"case_id": case_id, "title": case["title"], "case_status": case["case_status"], "created_at": case["created_at"]}

    **GET /api/cases** — list cases with optional status filter and simple pagination:
    - Query params: status: str | None = None, limit: int = 20, offset: int = 0
    - cases = await asyncio.to_thread(sqlite.list_investigation_cases, status)
    - paginated = cases[offset : offset + limit]
    - return {"cases": paginated, "total": len(cases), "limit": limit, "offset": offset}

    **GET /api/cases/{case_id}** — case detail:
    - case = await asyncio.to_thread(sqlite.get_investigation_case, case_id)
    - if case is None: raise HTTPException(404, "Case not found")
    - return case

    **PATCH /api/cases/{case_id}** — partial update:
    - updates = {k: v for k, v in body.model_dump().items() if v is not None}
    - if not updates: return {"case_id": case_id, "updated": False}
    - await asyncio.to_thread(sqlite.update_investigation_case, case_id, updates)
    - case = await asyncio.to_thread(sqlite.get_investigation_case, case_id)
    - return case

    **POST /api/cases/{case_id}/artifacts** — upload artifact:
    - Use `file: UploadFile = File(...)` and `description: str = Form("")`
    - Verify case exists first (404 if not)
    - content = await file.read()
    - data_dir = getattr(request.app.state, "data_dir", "data")
    - result = await save_artifact(data_dir, case_id, file.filename or "artifact", content, sqlite, description=description, mime_type=file.content_type)
    - Append artifact_id to the case's artifacts list: fetch case, append, update
    - return result

    **GET /api/cases/{case_id}/timeline** — reconstructed timeline:
    - duckdb = request.app.state.stores.duckdb
    - sqlite = request.app.state.stores.sqlite
    - Verify case exists (404 if not)
    - timeline = await build_timeline(case_id, duckdb, sqlite)
    - return {"case_id": case_id, "timeline": timeline, "total_events": len(timeline)}

    **POST /api/hunt** — execute hunt template:
    - duckdb = request.app.state.stores.duckdb
    - try: results = await execute_hunt(duckdb, body.template, body.params)
    - except ValueError as e: raise HTTPException(400, str(e))
    - return {"template": body.template, "params": body.params, "results": results, "result_count": len(results), "executed_at": datetime.now(timezone.utc).isoformat()}

    **GET /api/hunt/templates** — list templates:
    - templates = [{"name": t.name, "description": t.description, "param_keys": t.param_keys} for t in HUNT_TEMPLATES.values()]
    - return {"templates": templates}
