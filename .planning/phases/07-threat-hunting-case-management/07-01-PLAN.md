    class CaseManager:
        def create_investigation_case(
            self, sqlite_store: "SQLiteStore", title: str, description: str = "", case_id: str | None = None
        ) -> str:
            return sqlite_store.create_investigation_case(title, description, case_id)

        def get_investigation_case(

        ) -> dict | None:
            return sqlite_store.get_investigation_case(case_id)

            self, sqlite_store: "SQLiteStore", case_id: str, updates: dict
        ) -> None:
            sqlite_store.update_investigation_case(case_id, updates)
