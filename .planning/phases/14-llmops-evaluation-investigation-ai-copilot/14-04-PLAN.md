Implement merge_and_sort_timeline() as a pure Python function (no async, no store dependency — testable in isolation). The function signature accepts four source lists. edge_rows and playbook_rows have safe empty-list defaults so callers providing only events + detections continue to work without change:

```python
def merge_and_sort_timeline(
    event_rows: list[tuple],
    detection_rows: list[dict],
    edge_rows: list[dict] | None = None,
    playbook_rows: list[dict] | None = None,
) -> list[TimelineItem]:
    """
    Merge and sort timeline items from all sources.

    Args:
        event_rows:    Tuples from DuckDB normalized_events SELECT.
        detection_rows: Dicts from SQLiteStore.get_detections().
        edge_rows:      Dicts from SQLite graph_edges table. Defaults to empty list.
        playbook_rows:  Dicts from playbook_runs table (deferred — pass [] or None).
                        Accepted for forward-compatibility but always empty in Phase 14.
    """
    if edge_rows is None:
        edge_rows = []
    if playbook_rows is None:
        playbook_rows = []  # playbook_runs deferred to future phase

    items: list[TimelineItem] = []

    for row in event_rows:
        event_id, ts, event_type, severity, hostname, proc, atk_tech, atk_tac, cmd = row
        title = f"{event_type or 'Event'} on {hostname or 'unknown'}"
        if proc:
            title += f" — {proc}"
        entity_labels = [x for x in [hostname, proc] if x]
        items.append(TimelineItem(
            item_id=f"ev-{event_id}",
            item_type="event",
            timestamp=str(ts),
            title=title,
            severity=severity,
            attack_technique=atk_tech,
            attack_tactic=atk_tac,
            entity_labels=entity_labels,
            raw_id=event_id,
        ))

    for det in detection_rows:
        items.append(TimelineItem(
            item_id=f"det-{det['id']}",
            item_type="detection",
            timestamp=det.get("created_at", ""),
            title=det.get("rule_name") or "Detection",
            severity=det.get("severity"),
            attack_technique=det.get("attack_technique"),
            attack_tactic=det.get("attack_tactic"),
            entity_labels=[],
            raw_id=det["id"],
        ))

    for edge in edge_rows:
        # Edge dict keys depend on actual schema — try common names
        edge_id = edge.get("id") or edge.get("edge_id") or ""
        src = edge.get("src_label") or edge.get("src") or ""
        dst = edge.get("dst_label") or edge.get("dst") or ""
        ts_edge = edge.get("created_at") or edge.get("timestamp") or ""
        rel = edge.get("relationship") or edge.get("rel") or "related_to"
        items.append(TimelineItem(
            item_id=f"edge-{edge_id}",
            item_type="edge",
            timestamp=str(ts_edge),
            title=f"{src} —[{rel}]→ {dst}",
            severity=None,
            attack_technique=None,
            attack_tactic=None,
            entity_labels=[x for x in [src, dst] if x],
            raw_id=str(edge_id),
        ))

    # playbook_rows intentionally unused — deferred to future phase
    # When playbook_runs table is implemented, add loop here similar to edge_rows above.

    items.sort(key=lambda x: x.timestamp)
    return items
```

Implement a helper _fetch_edge_rows() that queries SQLite for graph edges related to the investigation. Entity names are extracted from detection rule_names (as a heuristic — rule names often contain hostname/process fragments). The query filters by src_label OR dst_label matching any entity name found in detections:

```python
def _get_entity_names_from_detections(detection_rows: list[dict]) -> list[str]:
    """Extract candidate entity labels from detection metadata for edge lookup."""
    names: set[str] = set()
    for det in detection_rows:
        for key in ("src_label", "dst_label", "hostname", "entity"):
            val = det.get(key)
            if val and isinstance(val, str):
                names.add(val)
    return list(names)
```

```python
def _get_edge_rows_sync(sqlite_store, entity_names: list[str]) -> list[dict]:
    """
    Fetch graph edges from SQLite where src_label or dst_label matches any entity name.
    Returns empty list if the graph_edges table does not exist or no entities to query.
    """
    if not entity_names:
        return []
    import sqlite3
    try:
        placeholders = ",".join("?" * len(entity_names))
        sql = (
            f"SELECT * FROM graph_edges "
            f"WHERE src_label IN ({placeholders}) OR dst_label IN ({placeholders}) "
            f"LIMIT 200"
        )
        params = entity_names + entity_names
        # Use the store's internal connection — check which pattern the store uses.
        # Pattern A: store has a _connect() context manager
        # Pattern B: store has a self._conn attribute
        # Read backend/stores/sqlite_store.py to determine the actual pattern.
        # If _connect() context manager exists:
        with sqlite_store._connect() as conn:
            rows = conn.execute(sql, params).fetchall()
            # Convert row_factory rows to dicts if not already
            if rows and not isinstance(rows[0], dict):
                cursor = conn.execute(f"PRAGMA table_info(graph_edges)")
                cols = [r[1] for r in cursor.fetchall()]
                return [dict(zip(cols, row)) for row in rows]
            return [dict(r) for r in rows]
    except Exception as exc:
        # Table may not exist or schema differs — safe fallback
        log.debug("graph_edges query skipped: %s", exc)
        return []
```

```python
@router.get("/{investigation_id}/timeline")
async def get_investigation_timeline(
    investigation_id: str,
    request: Request,
) -> JSONResponse:
    stores = request.app.state.stores

    # Fetch events from DuckDB
    event_rows = await stores.duckdb.fetch_all(
        """SELECT event_id, timestamp, event_type, severity, hostname, process_name,
                  attack_technique, attack_tactic, command_line
           FROM normalized_events
           WHERE case_id = ?
           ORDER BY timestamp ASC
           LIMIT 500""",
        (investigation_id,),
    )

    # Fetch detections from SQLite (sync call wrapped in to_thread)
    detection_rows: list[dict] = await asyncio.to_thread(
        stores.sqlite.get_detections, investigation_id, 200
    )

    # Extract entity names from detections to look up graph edges
    entity_names = _get_entity_names_from_detections(detection_rows)

    # Fetch graph edges from SQLite (sync, safe fallback if table absent)
    edge_rows: list[dict] = await asyncio.to_thread(
        _get_edge_rows_sync, stores.sqlite, entity_names
    )

    # playbook_rows: deferred to future phase — always empty in Phase 14
    # When playbook_runs table is implemented, add its fetch here.
    playbook_rows: list[dict] = []

    items = merge_and_sort_timeline(event_rows, detection_rows, edge_rows, playbook_rows)
    return JSONResponse({"items": [item.model_dump() for item in items], "total": len(items)})
```
