async def _write_telemetry(
    self, *, model: str, endpoint: str, prompt_chars: int,
    completion_chars: int, latency_ms: int, success: bool,
    error_type: Optional[str] = None,
) -> None:
    if self._duckdb_store is None:

    await self._duckdb_store.execute_write(
        """INSERT OR IGNORE INTO llm_calls
           (call_id, called_at, model, endpoint, prompt_chars,
            completion_chars, latency_ms, success, error_type)
           VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)""",
        (
            str(uuid.uuid4()),
            datetime.now(timezone.utc).isoformat(),
            model, endpoint, prompt_chars, completion_chars,
            latency_ms, success, error_type,
        ),
    )
```

async def _compute_llm_kpis(duckdb_store) -> tuple[dict[str, float], int, float]:
    rows = await duckdb_store.fetch_all(
        """SELECT model,
                  AVG(latency_ms)                  AS avg_latency,
                  COUNT(*)                          AS total,
                  SUM(CASE WHEN NOT success THEN 1 ELSE 0 END) AS errors
           FROM llm_calls
           GROUP BY model"""
    )
    if not rows:
        return {}, 0, 0.0
    avg_per_model = {row[0]: round(float(row[1] or 0), 1) for row in rows}
    total = sum(int(row[2]) for row in rows)
    errors = sum(int(row[3]) for row in rows)
    error_rate = round(errors / total, 4) if total > 0 else 0.0
    return avg_per_model, total, error_rate
```
