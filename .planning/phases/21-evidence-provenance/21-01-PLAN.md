    ```python
    import hashlib
    from pathlib import Path as _Path

    def _sha256_file(path: str) -> str:
        """Return SHA-256 hex digest of file contents. Runs in asyncio.to_thread."""
        h = hashlib.sha256()
        with open(path, "rb") as f:
            for chunk in iter(lambda: f.read(65536), b""):
                h.update(chunk)
        return h.hexdigest()
    ```
