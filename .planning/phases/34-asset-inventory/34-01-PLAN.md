    Tag extraction functions (module-level, not methods — imported by matcher.py and API):
    _TECH_RE = re.compile(r"^attack\.(t\d{4})(?:\.\d+)?$", re.IGNORECASE)

    def extract_attack_techniques_from_rule(rule: SigmaRule) -> list[str]:
        Return [m.group(1).upper() for tag in rule.tags if (m := _TECH_RE.match(tag.name))]

    def scan_rules_dir_for_coverage(rules_dir: Path) -> dict[str, list[str]]:
        Iterate rules_dir.rglob("*.yml"), SigmaRule.from_yaml(text), extract techniques, build {tech_id: [title]} dict. Wrap each file in try/except — skip malformed silently.
