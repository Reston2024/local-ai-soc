    learn_one(features: dict, verdict: str) -> None:
      label = 1 if verdict == "TP" else 0
      pred = self._model.predict_one(features)
      if pred is not None: self._metric.update(label, pred)

    upsert_feedback(self, detection_id: str, verdict: str) -> None:
      now = datetime.now(timezone.utc).isoformat()
      INSERT INTO feedback (id, detection_id, verdict, created_at, updated_at)
      VALUES (?, ?, ?, ?, ?)
      ON CONFLICT(detection_id) DO UPDATE SET verdict=excluded.verdict, updated_at=excluded.updated_at
      self._conn.commit()

    get_verdict_for_detection(self, detection_id: str) -> str | None:
      row = self._conn.execute("SELECT verdict FROM feedback WHERE detection_id = ?", (detection_id,)).fetchone()
      return row[0] if row else None

    get_feedback_stats(self) -> dict:
      row = self._conn.execute(
        "SELECT COUNT(*), SUM(CASE WHEN verdict='TP' THEN 1 ELSE 0 END), SUM(CASE WHEN verdict='FP' THEN 1 ELSE 0 END) FROM feedback"
      ).fetchone()
      total, tp, fp = row[0] or 0, row[1] or 0, row[2] or 0
      return {"verdicts_given": total, "tp_rate": tp/total if total else 0.0, "fp_rate": fp/total if total else 0.0}
