Add the route:
```python
@router.post("/{recommendation_id}/dispatch")
async def dispatch_recommendation(recommendation_id: str, request: Request) -> JSONResponse:
    """
    Validate and dispatch an approved recommendation artifact (ADR-030).

    Requires status == "approved". Re-validates artifact against recommendation.schema.json
    via RecommendationArtifact pydantic model. Does NOT make outbound HTTP calls to the
    firewall executor — dispatch is schema validation only (future phase handles execution).
    """
    stores = request.app.state.stores

    rows = await stores.duckdb.fetch_all(_SELECT_BY_ID, [recommendation_id])
    if not rows:
        raise HTTPException(status_code=404, detail="Recommendation not found")

    rec = _row_to_dict(rows[0])

    if rec.get("status") != "approved":
        return JSONResponse(
            status_code=409,
            content={
                "error": "not_approved",
                "detail": f"Recommendation status is '{rec.get('status')}'; must be 'approved' to dispatch.",
            },
        )

    try:
        artifact = RecommendationArtifact(**rec)
    except Exception as exc:
        # Catch both pydantic ValidationError and jsonschema.ValidationError
        import jsonschema
        from pydantic import ValidationError
        if isinstance(exc, (ValidationError, jsonschema.ValidationError)):
            detail = exc.errors() if hasattr(exc, "errors") else [str(exc)]
            return JSONResponse(
                status_code=422,
                content={"error": "schema_validation_failed", "detail": detail},
            )
        raise  # Unexpected error — propagate

    log.info(
        "Recommendation dispatched (schema validated)",
        recommendation_id=recommendation_id,
        artifact_type=rec.get("type"),
    )
    return JSONResponse(
        status_code=200,
        content={
            "dispatched": True,
            "recommendation_id": recommendation_id,
            "artifact_type": rec.get("type"),
        },
    )
```

```typescript
export interface DispatchResult {
  dispatched: boolean;
  recommendation_id: string;
  artifact_type: string;
}

export async function dispatchRecommendation(id: string): Promise<DispatchResult> {
  const res = await fetch(`/api/recommendations/${id}/dispatch`, {
    method: "POST",
    headers: { "Content-Type": "application/json", ...authHeaders() },
  });
  if (!res.ok) {
    const body = await res.json().catch(() => ({}));
    throw new Error(body.detail ?? body.error ?? `dispatch failed: ${res.status}`);
  }
  return res.json();
}
```

```svelte
<script lang="ts">
  import { dispatchRecommendation } from '$lib/api';

  let { recommendation } = $props();
  let dispatching = $state(false);
  let dispatched = $state(false);
  let dispatchError = $state<string | null>(null);

  async function handleDispatch() {
    dispatching = true;
    dispatchError = null;
    try {
      await dispatchRecommendation(recommendation.recommendation_id);
      dispatched = true;
    } catch (err) {
      dispatchError = err instanceof Error ? err.message : 'Dispatch failed';
    } finally {
      dispatching = false;
    }
  }
</script>

{#if recommendation.status === 'approved' && !dispatched}
  <button onclick={handleDispatch} disabled={dispatching}>
    {dispatching ? 'Dispatching...' : 'Dispatch'}
  </button>
{/if}
{#if dispatched}
  <span class="text-green-600">Dispatched</span>
{/if}
{#if dispatchError}
  <span class="text-red-600">{dispatchError}</span>
{/if}
```
