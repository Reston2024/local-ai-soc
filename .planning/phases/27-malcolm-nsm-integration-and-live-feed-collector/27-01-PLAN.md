   ssh opsadmin@192.168.1.22 "cd /opt/malcolm && python3 -c \"

txt = open('docker-compose.yml').read()
# Find opensearch service and add port if missing
if '9200:9200' not in txt:
    txt = re.sub(
        r'(services:\\s+opensearch[^:]*:.*?ports:\\s+)',
        r'\\1- \"9200:9200\"\\n      ',
        txt, flags=re.DOTALL
    )
    open('docker-compose.yml', 'w').write(txt)
    print('PATCHED')
else:
    print('ALREADY_MAPPED')
\""
