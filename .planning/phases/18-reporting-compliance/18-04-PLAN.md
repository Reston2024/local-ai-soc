def _severity_to_int(severity: str) -> int:

@router.get("/compliance")

**NIST CSF path (framework == "nist-csf"):**

1. Gather evidence via asyncio.to_thread (all synchronous SQLite queries):
   - detections = `SELECT id, rule_id, rule_name, severity, attack_technique, attack_tactic, created_at FROM detections ORDER BY created_at DESC LIMIT 500`
   - investigations = `SELECT case_id, title, case_status, created_at, updated_at, analyst_notes FROM investigation_cases ORDER BY created_at DESC LIMIT 200`
   - playbook_runs = `SELECT run_id, playbook_id, investigation_id, status, started_at, completed_at FROM playbook_runs ORDER BY started_at DESC LIMIT 200`
   - reports_list = `SELECT id, type, title, created_at FROM reports ORDER BY created_at DESC LIMIT 100`

2. Gather KPI evidence from DuckDB:
   - kpi_rows = await stores.duckdb.fetch_all("SELECT * FROM daily_kpi_snapshots ORDER BY snapshot_date DESC LIMIT 90")
   - If table doesn't exist (exception), set kpi_rows = []

3. Build evidence dict keyed by CSF function:
   ```python
   evidence = {
       "GOVERN":   {"kpi_snapshots": [dict(r) for r in kpi_rows]},
       "IDENTIFY": {"detections": detections, "technique_count": len({d["attack_technique"] for d in detections if d.get("attack_technique")})},
       "PROTECT":  {"playbook_runs_completed": [r for r in playbook_runs if r["status"] == "completed"]},
       "DETECT":   {"detections": detections, "rules_fired": len({d["rule_id"] for d in detections})},
       "RESPOND":  {"investigations": investigations, "playbook_runs": playbook_runs},
       "RECOVER":  {"investigations_resolved": [i for i in investigations if i["case_status"] in ("resolved","closed")], "playbook_runs_completed": [r for r in playbook_runs if r["status"] == "completed"]},
   }
   ```

4. Build summary HTML string:
   ```html
   <!DOCTYPE html><html><head><meta charset="utf-8">
   <title>NIST CSF 2.0 Evidence Package</title>
   <style>body{font-family:sans-serif;max-width:900px;margin:40px auto}table{border-collapse:collapse;width:100%}td,th{border:1px solid #ccc;padding:8px;text-align:left}th{background:#f0f0f0}</style>
   </head><body>
   <h1>NIST CSF 2.0 Evidence Package</h1>
   <p>Generated: {datetime.now(timezone.utc).isoformat()}</p>
   <table><tr><th>CSF Function</th><th>Evidence Items</th></tr>
   {rows_html}
   </table>
   </body></html>
   ```
   Where rows_html = one `<tr>` per function listing the item counts.

5. Build ZIP in memory using Python's zipfile module (no temp files):
   ```python
   import io, zipfile
   buf = io.BytesIO()
   with zipfile.ZipFile(buf, "w", zipfile.ZIP_DEFLATED) as zf:
       for func, data in evidence.items():
           zf.writestr(f"nist-csf/{func.lower()}.json", json.dumps(data, indent=2, default=str))
       zf.writestr("summary.html", summary_html)
   buf.seek(0)
   zip_bytes = buf.read()
   ```

6. Return Response with media_type="application/zip" and Content-Disposition header:
   ```python
   filename = f"nist-csf-evidence-{datetime.now(timezone.utc).strftime('%Y%m%d')}.zip"
   return Response(
       content=zip_bytes,
       media_type="application/zip",
       headers={"Content-Disposition": f'attachment; filename="{filename}"'},
   )
   ```

**TheHive path (framework == "thehive"):**

1. Fetch investigations from SQLite (same query as NIST path, selecting analyst_notes)
2. Build alerts list (TheHive Alert format per investigation).
   NOTE: investigation_cases has no "description" column — use analyst_notes for description content:
   ```python
   alerts = [
       {
           "title": inv["title"],
           "description": inv.get("analyst_notes", "") or "",
           "severity": 2,
           "tags": ["ai-soc-brain", inv["case_status"]],
           "source": "AI-SOC-Brain",
           "sourceRef": inv["case_id"],
           "type": "internal",
           "date": inv["created_at"],
       }
       for inv in investigations
   ]
   cases = [
       {
           "title": inv["title"],
           "description": inv.get("analyst_notes", "") or "",
           "severity": 2,
           "tags": ["ai-soc-brain"],
           "status": "Resolved" if inv["case_status"] in ("resolved","closed") else "Open",
           "startDate": inv["created_at"],
           "endDate": inv.get("updated_at"),
       }
       for inv in investigations
   ]
   ```
3. Build ZIP:
   ```python
   buf = io.BytesIO()
   with zipfile.ZipFile(buf, "w", zipfile.ZIP_DEFLATED) as zf:
       zf.writestr("thehive/alerts.json", json.dumps(alerts, indent=2, default=str))
       zf.writestr("thehive/cases.json", json.dumps(cases, indent=2, default=str))
   buf.seek(0)
   zip_bytes = buf.read()
   filename = f"thehive-export-{datetime.now(timezone.utc).strftime('%Y%m%d')}.zip"
   return Response(
       content=zip_bytes,
       media_type="application/zip",
       headers={"Content-Disposition": f'attachment; filename="{filename}"'},
   )
   ```
