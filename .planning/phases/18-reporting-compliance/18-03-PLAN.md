3. Add method to DuckDBStore class. IMPORTANT: DuckDB does NOT support `INSERT OR REPLACE` syntax — use the standard SQL upsert form:
```python
async def upsert_daily_kpi_snapshot(
    self,
    snapshot_date: str,      # "YYYY-MM-DD"
    mttd_minutes: float,
    mttr_minutes: float,
    mttc_minutes: float,
    alert_volume: int,
    false_positive_count: int,
    investigation_count: int,
    detection_count: int,
) -> None:
    """Upsert the daily KPI snapshot row for snapshot_date."""
    from datetime import datetime, timezone
    computed_at = datetime.now(timezone.utc).isoformat()
    await self.execute_write(
        """
        INSERT INTO daily_kpi_snapshots
          (snapshot_date, mttd_minutes, mttr_minutes, mttc_minutes,
           alert_volume, false_positive_count, investigation_count,
           detection_count, computed_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        ON CONFLICT (snapshot_date) DO UPDATE SET
            mttd_minutes         = EXCLUDED.mttd_minutes,
            mttr_minutes         = EXCLUDED.mttr_minutes,
            mttc_minutes         = EXCLUDED.mttc_minutes,
            alert_volume         = EXCLUDED.alert_volume,
            false_positive_count = EXCLUDED.false_positive_count,
            investigation_count  = EXCLUDED.investigation_count,
            detection_count      = EXCLUDED.detection_count,
            computed_at          = EXCLUDED.computed_at
        """,
        [snapshot_date, mttd_minutes, mttr_minutes, mttc_minutes,
         alert_volume, false_positive_count, investigation_count,
         detection_count, computed_at],
    )
```

Logic:
1. Parse `metric` query param (comma-separated string, e.g. "mttd,mttr"). Default: "mttd".
2. Parse `days` query param (int). Default: 30. Cap at 365.
3. Validate each metric name against METRIC_COLUMNS; return 400 if unknown metric.
4. Build SQL:
   ```sql
   SELECT snapshot_date, {col1}, {col2}, ...
   FROM daily_kpi_snapshots
   WHERE snapshot_date >= DATE('now', '-{days} days')
   ORDER BY snapshot_date ASC
   ```
   Use `await stores.duckdb.fetch_all(sql)` (no params needed since days is validated int — safe to format directly).
5. Build response:
   - Single metric → `{"metric": "mttd", "data": [{"date": "2026-01-01", "value": 12.5}, ...]}`
   - Multiple metrics → `{"mttd": [{"date": ..., "value": ...}], "mttr": [...]}`
6. If daily_kpi_snapshots table doesn't exist yet (DuckDB raises), catch exception and return empty data gracefully.

async def _take_daily_kpi_snapshot(stores) -> None:
    """Compute and upsert today's KPI snapshot. Called by APScheduler at midnight."""
    from datetime import date
    import asyncio
    try:
        svc = MetricsService(stores)
        snap = await svc.compute_all_kpis()
        today = date.today().isoformat()

        # investigation_count and detection_count are NOT on KpiSnapshot —
        # query SQLite directly using the store helper methods (asyncio.to_thread
        # because SQLiteStore methods are synchronous).
        inv_rows = await asyncio.to_thread(stores.sqlite.list_investigations)
        det_rows = await asyncio.to_thread(stores.sqlite.list_detections)
        inv_count = len(inv_rows) if inv_rows else 0
        det_count = len(det_rows) if det_rows else 0

        await stores.duckdb.upsert_daily_kpi_snapshot(
            snapshot_date=today,
            mttd_minutes=snap.mttd.value,
            mttr_minutes=snap.mttr.value,
            mttc_minutes=snap.mttc.value,
            alert_volume=int(snap.alert_volume_24h.value),
            false_positive_count=int(snap.false_positive_rate.value * max(snap.alert_volume_24h.value, 1)),
            investigation_count=inv_count,
            detection_count=det_count,
        )
        log.info("Daily KPI snapshot upserted", date=today)
    except Exception as exc:
        log.warning("Daily KPI snapshot failed: %s", exc)
```

NOTE: If `list_investigations` or `list_detections` do not exist on SQLiteStore, add them as minimal methods in backend/stores/sqlite_store.py:

def list_investigations(self) -> list[dict]:
    with self._conn() as conn:

        return [dict(r) for r in rows]

def list_detections(self) -> list[dict]:
    with self._conn() as conn:
        rows = conn.execute("SELECT id FROM detections").fetchall()
        return [dict(r) for r in rows]
```
