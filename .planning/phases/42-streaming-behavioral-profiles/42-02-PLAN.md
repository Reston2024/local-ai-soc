<objective>
Implement the River HalfSpaceTrees online ML scoring engine and wire it into the ingest pipeline so every event gets an anomaly_score at ingest time.

Purpose: Close the detection gap — events that don't match any Sigma rule but deviate from baseline behavior get scored. High scorers surface in Plan 03.

Output: AnomalyScorer class, model persistence to disk, anomaly_score column in DuckDB, scoring wired into IngestionLoader.

NOTE on serialization: River models are serialized using Python's pickle module. This is a well-established pattern for ML model persistence. The models are written and read exclusively by this process from a local directory (data/anomaly_models/). There is no deserialization of untrusted external content — the security risk is limited to local filesystem access, which is already within the application's trust boundary.
</objective>

    ```python
    """
    Online anomaly scoring using River HalfSpaceTrees.

    One model per entity_key (subnet_24, process_name). Models are persisted
    to ANOMALY_MODEL_DIR using Python pickle (local trusted data — written and
    read by this process only, never from untrusted external sources).
    Score + learn happen at ingest time (synchronous, called from asyncio.to_thread).
    """
    from __future__ import annotations
    import pickle  # noqa: S301 — local trusted data only, see module docstring
    import re
    from pathlib import Path
    from typing import Optional

    from river.anomaly import HalfSpaceTrees

    from backend.core.logging import get_logger

    log = get_logger(__name__)

    _DEFAULT_MODEL_DIR = "data/anomaly_models"
    _HST_N_TREES = 25
    _HST_HEIGHT = 15
    _HST_WINDOW = 250


    def entity_key(ip: Optional[str], process: Optional[str]) -> tuple[str, str]:
        """Return (subnet_24, process_name) peer-group key.
        Subnet is first 3 octets of IPv4. Falls back to 'unknown_subnet'."""
        proc = (process or "unknown").lower().strip()
        if ip:
            parts = ip.split(".")
            if len(parts) == 4:
                subnet = ".".join(parts[:3])
            else:
                subnet = "unknown_subnet"
        else:
            subnet = "unknown_subnet"
        return (subnet, proc)


    def _safe_filename(key: tuple[str, str]) -> str:
        """Convert entity key to a safe filename."""
        raw = f"{key[0]}__{key[1]}"
        return re.sub(r"[^a-zA-Z0-9._-]", "_", raw) + ".pkl"


    class AnomalyScorer:
        """Per-entity HalfSpaceTrees scorer with disk persistence."""

        def __init__(self, model_dir: str = _DEFAULT_MODEL_DIR) -> None:
            self._model_dir = Path(model_dir)
            self._model_dir.mkdir(parents=True, exist_ok=True)
            self._cache: dict[tuple, HalfSpaceTrees] = {}

        def _get_model(self, entity: tuple[str, str]) -> HalfSpaceTrees:
            if entity not in self._cache:
                loaded = self.load_model(entity)
                self._cache[entity] = loaded or HalfSpaceTrees(
                    n_trees=_HST_N_TREES,
                    height=_HST_HEIGHT,
                    window_size=_HST_WINDOW,
                )
            return self._cache[entity]

        def score_one(self, features: dict, entity: tuple[str, str]) -> float:
            model = self._get_model(entity)
            score = model.score_one(features)
            return float(max(0.0, min(1.0, score)))

        def learn_one(self, features: dict, entity: tuple[str, str]) -> None:
            model = self._get_model(entity)
            model.learn_one(features)

        def save_model(self, entity: tuple[str, str]) -> None:
            if entity not in self._cache:
                return
            path = self._model_dir / _safe_filename(entity)
            try:
                with open(path, "wb") as f:
                    pickle.dump(self._cache[entity], f)  # noqa: S301
            except Exception as exc:
                log.warning("Failed to save anomaly model", entity=entity, error=str(exc))

        def load_model(self, entity: tuple[str, str]) -> Optional[HalfSpaceTrees]:
            path = self._model_dir / _safe_filename(entity)
            if not path.exists():
                return None
            try:
                with open(path, "rb") as f:
                    return pickle.load(f)  # noqa: S301 — local trusted data
            except Exception as exc:
                log.warning("Failed to load anomaly model", entity=entity, error=str(exc))
                return None
    ```
