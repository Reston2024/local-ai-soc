    ```yaml
    name: CI

    on:
      push:
        branches: ["main", "feature/**", "fix/**"]
      pull_request:
        branches: ["main"]

    jobs:
      lint:
        name: Lint (ruff)
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-python@v5
            with:
              python-version: "3.12"
          - name: Install uv
            run: pip install uv
          - name: Install dependencies
            run: uv sync
          - name: Run ruff
            run: uv run ruff check .

      test:
        name: Test (pytest + coverage)
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-python@v5
            with:
              python-version: "3.12"
          - name: Install uv
            run: pip install uv
          - name: Install dependencies (with dev extras for pytest-cov)
            run: uv sync --extra dev
          - name: Run tests with coverage
            run: |
              uv run pytest tests/unit/ tests/security/ \
                --cov=backend \
                --cov=ingestion \
                --cov=detections \
                --cov-report=xml:coverage.xml \
                --cov-report=term-missing \
                --cov-fail-under=70 \
                --junitxml=junit.xml \
                -q
          - name: Upload test results
            uses: actions/upload-artifact@v4
            if: always()
            with:
              name: test-results
              path: |
                junit.xml
                coverage.xml

      dependency-audit:
        name: Dependency Audit (pip-audit)
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-python@v5
            with:
              python-version: "3.12"
          - name: Install uv
            run: pip install uv
          - name: Install dependencies
            run: uv sync
          - name: Install pip-audit
            run: uv run pip install pip-audit
          - name: Run pip-audit
            run: uv run pip-audit --desc

      secret-scan:
        name: Secret Scan (gitleaks)
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
            with:
              fetch-depth: 0
          - name: Run gitleaks
            uses: gitleaks/gitleaks-action@v2
            env:
              GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
    ```

    Notes on implementation choices (per Claude's Discretion from CONTEXT.md):
    - pip-audit chosen over trivy (simpler, Python-native, no auth required)
    - gitleaks chosen over trufflehog (single binary, no auth required via gitleaks-action)
    - uv for dependency management (consistent with project conventions)
    - Smoke tests NOT included (PowerShell-only, require running Windows backend — per locked decision)
    - Coverage threshold: 70% (per locked decision)
    - JUnit XML + coverage XML uploaded as artifacts (per locked decision)
