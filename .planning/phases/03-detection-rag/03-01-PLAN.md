```python
def test_try_index_attempts_http_when_url_set(self):
    """try_index must attempt HTTP PUT when OPENSEARCH_URL is configured."""
    import backend.src.ingestion.opensearch_sink as sink
    with patch.dict("os.environ", {"OPENSEARCH_URL": "http://localhost:9200"}):
        # Reset the module-level OPENSEARCH_URL (it's read at import time)
        original_url = sink.OPENSEARCH_URL
        sink.OPENSEARCH_URL = "http://localhost:9200"
        try:
            with patch.object(sink, "_get_client") as mock_client:
                mock_http = MagicMock()
                mock_http.put.return_value = MagicMock(status_code=201)
                mock_client.return_value = mock_http
                event = NormalizedEvent(id="test-1", timestamp="2026-03-15T00:00:00Z",
                                        host="testhost", event_type="dns")
                result = sink.try_index(event)
                assert result is True
                mock_http.put.assert_called_once()
        finally:
            sink.OPENSEARCH_URL = original_url
```
