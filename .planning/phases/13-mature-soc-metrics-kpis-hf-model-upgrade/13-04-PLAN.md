    **MetricsService class:**
    ```python
    class MetricsService:
        def __init__(self, stores: Stores) -> None:
            self._stores = stores

        async def compute_mttd(self) -> KpiValue:
            """
            MTTD = avg(first_detection.created_at - min(event.timestamp))
            for all detections. Returns 0.0 if no detections.
            Queries: detections.created_at vs normalized_events.timestamp
            Uses asyncio.to_thread for SQLite; await store.fetch_all for DuckDB.
            """

        async def compute_mttr(self) -> KpiValue:
            """
            MTTR = avg(case.updated_at - case.created_at) for closed cases.
            Proxy: avg minutes between case created_at and last detection in that case.
            Returns 0.0 if no closed cases.
            """

        async def compute_mttc(self) -> KpiValue:
            """
            MTTC = avg(case.closed_at - case.created_at) for all cases with status != 'active'.
            Returns 0.0 if no closed cases. Uses same cases table.
            """

        async def compute_false_positive_rate(self) -> KpiValue:
            """
            FP rate = count(detections with severity='low' and no case_id) / count(all detections).
            Proxy metric (true FP tracking requires analyst feedback — not yet implemented).
            Returns 0.0 if no detections.
            """

        async def compute_alert_volume(self) -> KpiValue:
            """Count of detections with created_at >= NOW() - 24h from SQLite."""

        async def compute_active_rules(self) -> KpiValue:
            """Count of distinct rule_ids in detections table (rules that have fired)."""

        async def compute_open_cases(self) -> KpiValue:
            """Count of cases with status='active' from SQLite."""

        async def compute_assets_monitored(self) -> KpiValue:
            """Count of distinct non-null hostnames from DuckDB normalized_events."""

        async def compute_log_sources(self) -> KpiValue:
            """Count of distinct non-null source_type values from DuckDB normalized_events."""

        async def compute_all_kpis(self) -> KpiSnapshot:
            """Run all compute_* coroutines concurrently via asyncio.gather, return KpiSnapshot."""
    ```

    ```python
    """
    GET /api/metrics/kpis — returns cached SOC KPI snapshot.

    APScheduler AsyncIOScheduler recomputes every 60 seconds.
    Endpoint returns the last cached value immediately (< 1ms response time).
    """
    from __future__ import annotations

    import asyncio
    from datetime import datetime, timezone
    from typing import Optional

    from apscheduler.schedulers.asyncio import AsyncIOScheduler
    from fastapi import APIRouter, Request
    from fastapi.responses import JSONResponse

    from backend.core.deps import StoresDep
    from backend.services.metrics_service import KpiSnapshot, MetricsService

    router = APIRouter(tags=["metrics"])

    # Module-level cache — shared across requests in the single-worker process
    _kpi_cache: Optional[KpiSnapshot] = None
    _scheduler: Optional[AsyncIOScheduler] = None


    async def _refresh_kpis(stores) -> None:
        global _kpi_cache
        try:
            svc = MetricsService(stores)
            _kpi_cache = await svc.compute_all_kpis()
        except Exception as exc:
            # Log but never crash the scheduler job
            import logging
            logging.getLogger(__name__).warning("KPI refresh failed: %s", exc)


    @router.get("/metrics/kpis")
    async def get_kpis(request: Request, stores: StoresDep) -> JSONResponse:
        """
        Return the most recent SOC KPI snapshot.

        The scheduler refreshes the cache every 60 seconds.
        On the first call (cache cold) the KPIs are computed inline.
        """
        global _kpi_cache, _scheduler

        # Start scheduler on first call (attaches to running event loop)
        if _scheduler is None:
            _scheduler = AsyncIOScheduler()
            # Capture stores reference for the scheduler job via closure
            _scheduler.add_job(
                _refresh_kpis,
                "interval",
                seconds=60,
                args=[stores],
                id="kpi_refresh",
                replace_existing=True,
            )
            _scheduler.start()

        # Cold cache: compute inline
        if _kpi_cache is None:
            await _refresh_kpis(stores)

        if _kpi_cache is None:
            # Fallback if compute still failed
            return JSONResponse(status_code=503, content={"detail": "KPI data unavailable"})

        return JSONResponse(content=_kpi_cache.model_dump(mode="json"))
    ```
